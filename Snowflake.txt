CREATE DATABASE SMART_CITY_DB;
CREATE SCHEMA SMART_CITY_DB.TRAFFIC;
USE SCHEMA SMART_CITY_DB.TRAFFIC;

-- Target 1
CREATE TABLE TRANSPORT_ROUTES (
    ROUTE_ID          NUMBER,
    ROUTE_NAME        VARCHAR(100),
    ROUTE_TYPE        VARCHAR(50),
    CITY              VARCHAR(100),
    OPERATIONAL_STATUS VARCHAR(20),
    LOAD_TIMESTAMP    TIMESTAMP DEFAULT CURRENT_TIMESTAMP()
);

-- Target 2
CREATE TABLE VEHICLES (
    VEHICLE_ID     NUMBER,
    VEHICLE_NUMBER VARCHAR(50),
    VEHICLE_TYPE   VARCHAR(50),
    ASSIGNED_ROUTE NUMBER,
    VEHICLE_STATUS VARCHAR(20),
    LOAD_TIMESTAMP TIMESTAMP DEFAULT CURRENT_TIMESTAMP()
);

-- Target 3
CREATE TABLE TRAFFIC_RECORDS (
    TRAFFIC_ID      NUMBER,
    ROUTE_ID        NUMBER,
    CONGESTION_LEVEL NUMBER,
    RECORDED_AT     TIMESTAMP,
    TRAFFIC_STATUS  VARCHAR(20),
    IS_HEAVY_FLAG   VARCHAR(3),   -- YES/NO derived field
    LOAD_TIMESTAMP  TIMESTAMP DEFAULT CURRENT_TIMESTAMP()
);

-- Target 4
CREATE TABLE PASSENGER_TRIPS (
    TRIP_ID        NUMBER,
    VEHICLE_ID     NUMBER,
    PASSENGER_COUNT NUMBER,
    TRIP_DATE      DATE,
    TRIP_STATUS    VARCHAR(20),
    LOAD_TIMESTAMP TIMESTAMP DEFAULT CURRENT_TIMESTAMP()
);
-- Target 5
CREATE TABLE TRAFFIC_MAINTAIN (
    ALERT_ID VARCHAR(10),
    ROUTE_ID VARCHAR(10),
    ALERT_REASON VARCHAR(100),
    ALERT_SEVERITY VARCHAR(20),
    ALERT_STATUS VARCHAR(20)
);
